# Environment Variables for Cloudflare D1 Integration

# Cloudflare Account ID
# Find this in your Cloudflare dashboard under Account > Workers & Pages
NEXT_PUBLIC_CLOUDFLARE_ACCOUNT_ID=your_account_id_here

# Cloudflare D1 Database ID
# Find this in your Cloudflare dashboard under Workers & Pages > D1
NEXT_PUBLIC_CLOUDFLARE_DATABASE_ID=your_database_id_here

# Cloudflare API Token
# Create this in your Cloudflare dashboard under My Profile > API Tokens
# Required permissions: Account.D1 (Edit)
NEXT_PUBLIC_CLOUDFLARE_API_TOKEN=your_api_token_here
